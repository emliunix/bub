\section{Overview}

System SB (Syntax-directed Bidirectional type system with visible type application) is defined in Eisenberg et al.\ (2016) \textit{``Visible Type Application''} (ESOP 2016, extended version). It is a syntax-directed bidirectional type system combining:
\begin{itemize}
\item Higher-rank polymorphism (predicative)
\item Visible type application (@)
\item Scoped type variables
\item Bidirectional type checking (synthesis vs. checking)
\end{itemize}

The system uses \textbf{lazy instantiation}: polymorphic types are not eagerly instantiated at variable lookup, but delayed until application sites.

\section{Judgments}

System SB has \textbf{five mutually recursive judgments}:

\begin{enumerate}
\item $\Gamma \vdash_{\mathrm{sb}} e \Rightarrow \phi$ --- Synthesize monotype
\item $\Gamma \vdash^*_{\mathrm{sb}} e \Rightarrow \upsilon$ --- Synthesize specified polytype
\item $\Gamma \vdash^{\mathit{gen}}_{\mathrm{sb}} e \Rightarrow \sigma$ --- Synthesize generalized polytype
\item $\Gamma \vdash_{\mathrm{sb}} e \Leftarrow \rho$ --- Check against rho-type
\item $\Gamma \vdash^*_{\mathrm{sb}} e \Leftarrow \upsilon$ --- Check against specified polytype
\end{enumerate}

\begin{figure}[p]
\centering
\textbf{Monotype synthesis:} $\boxed{\Gamma \vdash_{\mathrm{sb}} e \Rightarrow \phi}$
\[
\frac{\Gamma, x\!:\!\tau \vdash_{\mathrm{sb}}^{*} e \Rightarrow \upsilon}{\Gamma \vdash_{\mathrm{sb}} \lambda x.\, e \Rightarrow \tau \to \upsilon}\textsc{(SB\_Abs)}
\quad
\frac{}{\Gamma \vdash_{\mathrm{sb}} n \Rightarrow \mathit{Int}}\textsc{(SB\_Int)}
\]
\[
\frac{\Gamma \vdash_{\mathrm{sb}}^{*} e \Rightarrow \forall \bar{a}.\, \phi \quad \text{no other rule matches}}{\Gamma \vdash_{\mathrm{sb}} e \Rightarrow \phi[\bar{\tau}/\bar{a}]}\textsc{(SB\_InstS)}
\]

\vspace{1em}
\textbf{Specified polytype synthesis:} $\boxed{\Gamma \vdash_{\mathrm{sb}}^{*} e \Rightarrow \upsilon}$
\[
\frac{x\!:\!\forall\{\bar{a}\}.\, \upsilon \in \Gamma}{\Gamma \vdash_{\mathrm{sb}}^{*} x \Rightarrow \upsilon[\bar{\tau}/\bar{a}]}\textsc{(SB\_Var)}
\quad
\frac{\Gamma \vdash_{\mathrm{sb}} e_1 \Rightarrow \upsilon_1 \to \upsilon_2 \quad \Gamma \vdash_{\mathrm{sb}}^{*} e_2 \Leftarrow \upsilon_1}{\Gamma \vdash_{\mathrm{sb}}^{*} e_1\, e_2 \Rightarrow \upsilon_2}\textsc{(SB\_App)}
\]
\[
\frac{\Gamma \vdash \tau \quad \Gamma \vdash_{\mathrm{sb}}^{*} e \Rightarrow \forall a.\, \upsilon}{\Gamma \vdash_{\mathrm{sb}}^{*} e\, @\tau \Rightarrow \upsilon[\tau/a]}\textsc{(SB\_TApp)}
\quad
\frac{\Gamma \vdash \upsilon \quad \upsilon = \forall \bar{a}, \bar{b}.\, \phi \quad \Gamma, \bar{a} \vdash_{\mathrm{sb}}^{*} e \Leftarrow \phi \quad \bar{a}, \bar{b} \notin \mathrm{ftv}(\Gamma)}{\Gamma \vdash_{\mathrm{sb}}^{*} (\Lambda \bar{a}.\, e : \upsilon) \Rightarrow \upsilon}\textsc{(SB\_Annot)}
\]
\[
\frac{\Gamma \vdash_{\mathrm{sb}}^{\mathit{gen}} e_1 \Rightarrow \sigma_1 \quad \Gamma, x\!:\!\sigma_1 \vdash_{\mathrm{sb}}^{*} e_2 \Rightarrow \upsilon_2}{\Gamma \vdash_{\mathrm{sb}}^{*} \mathbf{let}\; x = e_1 \;\mathbf{in}\; e_2 \Rightarrow \upsilon_2}\textsc{(SB\_Let)}
\quad
\frac{\Gamma \vdash_{\mathrm{sb}} e \Rightarrow \phi \quad \text{no other rule matches}}{\Gamma \vdash_{\mathrm{sb}}^{*} e \Rightarrow \phi}\textsc{(SB\_Phi)}
\]

\vspace{1em}
\textbf{Generalization:} $\boxed{\Gamma \vdash_{\mathrm{sb}}^{\mathit{gen}} e \Rightarrow \sigma}$
\[
\frac{\Gamma \vdash_{\mathrm{sb}}^{*} e \Rightarrow \upsilon \quad \bar{a} = \mathrm{ftv}(\upsilon) \setminus \mathrm{ftv}(\Gamma)}{\Gamma \vdash_{\mathrm{sb}}^{\mathit{gen}} e \Rightarrow \forall\{\bar{a}\}.\, \upsilon}\textsc{(SB\_Gen)}
\]

\vspace{1em}
\textbf{Checking:} $\boxed{\Gamma \vdash_{\mathrm{sb}} e \Leftarrow \rho}$
\[
\frac{\Gamma \vdash_{\mathrm{sb}}^{\mathit{gen}} e_1 \Rightarrow \sigma_1 \quad \Gamma, x\!:\!\sigma_1 \vdash_{\mathrm{sb}} e_2 \Leftarrow \rho_2}{\Gamma \vdash_{\mathrm{sb}} \mathbf{let}\; x = e_1 \;\mathbf{in}\; e_2 \Leftarrow \rho_2}\textsc{(SB\_DLet)}
\quad
\frac{\Gamma, x\!:\!\upsilon_1 \vdash_{\mathrm{sb}}^{*} e \Leftarrow \rho_2}{\Gamma \vdash_{\mathrm{sb}} \lambda x.\, e \Leftarrow \upsilon_1 \to \rho_2}\textsc{(SB\_DAbs)}
\]
\[
\frac{\Gamma \vdash_{\mathrm{sb}}^{*} e \Rightarrow \upsilon_1 \quad \upsilon_1 \leq_{\mathrm{dsk}} \rho_2 \quad \text{no other rule matches}}{\Gamma \vdash_{\mathrm{sb}} e \Leftarrow \rho_2}\textsc{(SB\_Infer)}
\]

\vspace{1em}
\textbf{Specified polytype checking:} $\boxed{\Gamma \vdash_{\mathrm{sb}}^{*} e \Leftarrow \upsilon}$
\[
\frac{\mathrm{prenex}(\upsilon) = \forall \bar{a}.\, \rho \quad \bar{a} \notin \mathrm{ftv}(\Gamma) \quad \Gamma \vdash_{\mathrm{sb}} e \Leftarrow \rho}{\Gamma \vdash_{\mathrm{sb}}^{*} e \Leftarrow \upsilon}\textsc{(SB\_DeepSkol)}
\]

\textbf{Higher-rank instantiation:} $\boxed{\sigma_1 \leq_b \sigma_2}$
\[
\frac{}{\tau \leq_b \tau}\textsc{(B\_Refl)}
\quad
\frac{\upsilon_3 \leq_b \upsilon_1 \quad \upsilon_2 \leq_b \upsilon_4}{\upsilon_1 \to \upsilon_2 \leq_b \upsilon_3 \to \upsilon_4}\textsc{(B\_Fun)}
\]

\frac{\phi_1[\bar{\tau}/\bar{b}] \leq_b \phi_2}{\forall \bar{a}, \bar{b}.\, \phi_1 \leq_b \forall \bar{a}.\, \phi_2}\textsc{(B\_InstS)}
\quad
\frac{\upsilon_1[\bar{\tau}/\bar{a}] \leq_b \upsilon_2 \quad \bar{b} \notin \mathrm{ftv}(\forall\{\bar{a}\}.\, \upsilon_1)}{\forall\{\bar{a}\}.\, \upsilon_1 \leq_b \forall\{\bar{b}\}.\, \upsilon_2}\textsc{(B\_InstG)}
\]

\vspace{1em}
\textbf{Deep skolemization:} $\boxed{\sigma_1 \leq_{\mathrm{dsk}} \upsilon_2}$
\[
\frac{\begin{array}{c}
\mathit{prenex}(\upsilon_2) = \forall \bar{c}.\, \rho_2 \\
\phi_1[\bar{\tau}/\bar{a}][\bar{\tau}'/\bar{b}] \leq_{\mathrm{dsk}}^{*} \rho_2
\end{array}}{\forall\{\bar{a}\}, \bar{b}.\, \phi_1 \leq_{\mathrm{dsk}} \upsilon_2}\textsc{(DSK\_Inst)}
\]

\vspace{0.5em}
\textbf{Subsumption (auxiliary):} $\boxed{\phi_1 \leq_{\mathrm{dsk}}^{*} \rho_2}$
\[
\frac{}{\tau \leq_{\mathrm{dsk}}^{*} \tau}\textsc{(DSK\_Refl)}
\quad
\frac{\upsilon_3 \leq_{\mathrm{dsk}} \upsilon_1 \quad \upsilon_2 \leq_{\mathrm{dsk}} \rho_4}{\upsilon_1 \to \upsilon_2 \leq_{\mathrm{dsk}}^{*} \upsilon_3 \to \rho_4}\textsc{(DSK\_Fun)}
\]

\vspace{0.5em}
Define $\mathit{prenex}(\upsilon) = \forall \bar{a}.\, \rho$ as:
\[
\begin{array}{rcl}
\mathit{prenex}(\forall \bar{a}.\, \tau) & = & \forall \bar{a}.\, \tau \\
\mathit{prenex}(\forall \bar{a}.\, \upsilon_1 \to \upsilon_2) & = & \forall \bar{a}, \bar{b}.\, \upsilon_1 \to \rho_2 \quad \text{where } \forall \bar{b}.\, \rho_2 = \mathit{prenex}(\upsilon_2)
\end{array}
\]

\section{Key Differences from Peyton Jones 2007}

\begin{enumerate}
\item \textbf{Lazy Instantiation}: Polymorphic types are not instantiated at variable lookup, but delayed until application.

\item \textbf{Multiple Judgments}: Five mutually recursive judgments instead of one, handling monotypes, specified polytypes, and generalized polytypes separately.

\item \textbf{Visible Type Application}: New form $e \,@\,\tau$ with rule SB\_TApp allows explicit type instantiation.

\item \textbf{Scoped Type Variables}: Type variables in annotations scope over the annotated expression (SB\_Annot).

\item \textbf{Two Subsumption Relations}: $\leq_b$ for synthesis/checking boundary and $\leq_{\mathrm{dsk}}$ for checking against specified types.
\end{enumerate}
